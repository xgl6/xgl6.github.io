    /**
     * åˆ—å‡ºæ‰€æœ‰å¯¼å‡ºçš„æ•°æ®
     *
     * @return
     */
    @Override
    public List<BackupDTO> listExportedData() {

        String path = ServiceUtils.getProjectPathUpper();

        Path backupParentPath = Paths.get(path + "/public/backup");
        if (Files.notExists(backupParentPath)) {
            return Collections.emptyList();
        }

        // Build backup dto
        try (Stream<Path> subPathStream = Files.list(backupParentPath)) {
            List<BackupDTO> collect = subPathStream
                    .filter(backupPath -> StringUtils.startsWithIgnoreCase(backupPath.getFileName().toString(), SystemConst.BACKUP_PREFIX))
                    .map(backupPath -> buildBackupDto("/backup", backupPath))
                    //.sorted(Comparator.comparingLong(BackupDTO::getUpdateTime).reversed())
                    .collect(Collectors.toList());
            CollUtil.sort(collect, (o1, o2) -> DateUtil.compare(o2.getUpdateTime(), o1.getUpdateTime()));
            return collect;
        } catch (IOException e) {
            throw new RuntimeException("Failed to fetch backups", e);
        }
    }

    /**
     * æ„å»ºå¤‡ä»½æ–‡ä»¶
     *
     * @param basePath   å¤‡ä»½å‰ç¼€
     * @param backupPath æ–‡ä»¶å
     * @return
     */
    private BackupDTO buildBackupDto(String basePath, Path backupPath) {
        Assert.notNull(basePath, "Base path must not be null");
        Assert.notNull(backupPath, "Backup path must not be null");

        String backupFileName = backupPath.getFileName().toString();
        BackupDTO backup = new BackupDTO();
        try {
            String backupUri = basePath + "/" + backupFileName;
            backup.setDownloadLink(backupUri);
            backup.setFilename(backupFileName);
            backup.setUpdateTime(DateUtil.date(Files.getLastModifiedTime(backupPath).toMillis()));
            backup.setFileSize(Files.size(backupPath));
        } catch (IOException e) {
            throw new RuntimeException("Failed to access file " + backupPath, e);
        }

        return backup;
    }

    @Override
    public void importData(MultipartFile file) throws IOException {

         * 1.å…ˆéªŒè¯æ˜¯å¦æ˜¯zipå‹ç¼©åŒ…

        UserInfo userInfo = SystemUtil.getUserInfo();
        String tempDir = userInfo.getTempDir();
        int code = IdUtil.fastSimpleUUID().hashCode();

        //3.è§£å‹åˆ°æ–‡ä»¶åç›¸åŒçš„ç›®å½•ä¸­,ä¹Ÿå°±æ˜¯ /tmp/lims/upload/2020-09-22/UUID/backup/**
        ZipUtil.unzip(tempPath);
        //3.1 å¯¹tempPathè¿›è¡Œå¤„ç†,å»æ‰æ–‡ä»¶ååç¼€
        String subBefore = StrUtil.subBefore(tempPath, ".zip", true);
        //3.2 ä¿®æ­£è·¯å¾„
        subBefore = FileUtil.normalize(subBefore);

        //4.åˆ¤æ–­æ–‡ä»¶çš„å†…å®¹æ˜¯å¦ç¬¦åˆè¦æ±‚,ä¹Ÿå°±æ˜¯ä¸€ä¸ªsqlæ–‡ä»¶,ä¸€ä¸ªfilesæ–‡ä»¶å¤¹
        boolean sqlExist = FileUtil.exist(subBefore + "/backup.sql");

            FileUtil.del(tempDir + "/lims/upload/");
            throw new RuntimeException("ä¸Šä¼ çš„zipå‹ç¼©åŒ…é‡Œæ²¡æœ‰sqlæ–‡ä»¶");
        }
        boolean filesExist = FileUtil.exist(subBefore + "/files");
        if (!filesExist) {
            FileUtil.del(tempDir + "/lims/upload/");
            throw new RuntimeException("ä¸Šä¼ çš„zipå‹ç¼©åŒ…é‡Œæ²¡æœ‰filesæ–‡ä»¶å¤¹");
        }
        //5.å¼€å§‹å¯¼å…¥SQL
        String sh = "";
        String option = "";
        StrBuilder strBuilder = StrBuilder.create();
        OsInfo osInfo = SystemUtil.getOsInfo();
        if (osInfo.isWindows()) {
            //5.1 Windowsç¯å¢ƒä¸‹çš„å¯¼å…¥
            sh = "cmd";
            option = "/c";
            strBuilder.append("mysql -h127.0.0.1 -uroot -proot ym_lims < ");
        } else if (osInfo.isLinux()) {
            //5.2 Linuxç¯å¢ƒä¸‹çš„å¯¼å…¥
            sh = "/bin/sh";
            option = "-c";
            //ç”±äºMySQL 5.7 ç‰ˆæœ¬åšäº†é™åˆ¶  éœ€è¦åœ¨my.cnfé…ç½®æ–‡ä»¶é‡Œé…ç½®å¯†ç

        } else {
            throw new RuntimeException("ä¸æ”¯æŒçš„ç³»ç»Ÿç±»å‹,å¯¼å…¥ç»ˆæ­¢");
        }

        strBuilder.append("/backup.sql");
        String[] command = {sh, option, strBuilder.toString()};
        log.info("command:{}", Arrays.toString(command));
        Process exec = RuntimeUtil.exec(command);
        List<String> resultLines = RuntimeUtil.getResultLines(exec);

            log.info("resultLine:{}",resultLine);
        }

        //6.ç§»åŠ¨æ–‡ä»¶
        String filesPath = subBefore + "/files";
        String projectPath = ServiceUtils.getProjectPathUpper();
        projectPath = projectPath + "/public/";
        FileUtil.move(FileUtil.file(filesPath),FileUtil.file(projectPath),true);

        //7.åˆ é™¤äº§ç”Ÿçš„ä¸´æ—¶æ–‡ä»¶
        String delPath = "{}/lims/upload/{}/{}/";
        delPath = StrUtil.format(delPath,tempDir,DateUtil.today(),code);
        FileUtil.del(delPath);


    }
